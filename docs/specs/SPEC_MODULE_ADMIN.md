#### 6.3 Calcul Quality Score

**Formule** :
```typescript
function calculateQualityScore(textile: Textile): number {
  let score = 0;
  let maxScore = 0;
  
  // Champs critiques (10 points chacun)
  const criticalFields = [
    'name', 'material_type', 'color', 
    'price_value', 'quantity_value'
  ];
  
  criticalFields.forEach(field => {
    maxScore += 10;
    if (textile[field]) score += 10;
  });
  
  // Champs importants (5 points chacun)
  const importantFields = [
    'pattern', 'weave', 'width_value', 
    'weight_value', 'composition'
  ];
  
  importantFields.forEach(field => {
    maxScore += 5;
    if (textile[field]) score += 5;
  });
  
  // Champs utiles (2 points chacun)
  const usefulFields = [
    'description', 'image_url', 
    'certifications', 'supplier_name'
  ];
  
  usefulFields.forEach(field => {
    maxScore += 2;
    if (textile[field]) score += 2;
  });
  
  return Math.round((score / maxScore) * 100);
}
```
