### Logique de Calcul

// src/features/journey/services/yardageCalculator.ts

export interface CalculationParams {
  garments: GarmentConfig[];
  fabricWidth: number;      // cm
  marginPercent: number;    // 5, 10, 15
  fabricModifiers: {
    directional: boolean;   // +10%
    patternMatch: boolean;  // +20%
    velvet: boolean;        // +10%
    stretch: boolean;       // -10%
  };
}

export interface CalculationResult {
  totalYardage: number;
  recommendedYardage: number;  // Rounded up to 0.5m
  breakdown: GarmentBreakdown[];
  subtotal: number;
  marginAmount: number;
  modifiersAmount: number;
}

export function calculateYardage(params: CalculationParams): CalculationResult {
  let subtotal = 0;
  const breakdown: GarmentBreakdown[] = [];
  
  for (const garment of params.garments) {
    const config = GARMENT_TYPES[garment.type];
    let base = config.baseYardage[garment.size];
    
    // Apply variation modifiers
    if (garment.variations) {
      for (const [key, value] of Object.entries(garment.variations)) {
        if (config.modifiers[key]?.[value]) {
          base += config.modifiers[key][value];
        }
      }
    }
    
    // Adjust for fabric width
    if (params.fabricWidth < 140) {
      base *= 1.1;  // +10% for narrow fabric
    } else if (params.fabricWidth > 150) {
      base *= 0.95; // -5% for wide fabric
    }
    
    const garmentTotal = base * garment.quantity;
    subtotal += garmentTotal;
    
    breakdown.push({
      garment,
      baseYardage: base,
      quantity: garment.quantity,
      total: garmentTotal
    });
  }
  
  // Apply fabric modifiers
  let modifiersMultiplier = 1;
  if (params.fabricModifiers.directional) modifiersMultiplier += 0.1;
  if (params.fabricModifiers.patternMatch) modifiersMultiplier += 0.2;
  if (params.fabricModifiers.velvet) modifiersMultiplier += 0.1;
  if (params.fabricModifiers.stretch) modifiersMultiplier -= 0.1;
  
  const afterModifiers = subtotal * modifiersMultiplier;
  const modifiersAmount = afterModifiers - subtotal;
  
  // Apply safety margin
  const marginAmount = afterModifiers * (params.marginPercent / 100);
  const totalYardage = afterModifiers + marginAmount;
  
  // Round up to nearest 0.5m
  const recommendedYardage = Math.ceil(totalYardage * 2) / 2;
  
  return {
    totalYardage,
    recommendedYardage,
    breakdown,
    subtotal,
    marginAmount,
    modifiersAmount
  };
}
